Nim Code Examples

User Input

Reading user input from the terminal:

# greeting.nim
echo "What's your name?"
let name = readLine(stdin)
echo "Hello, ", name, "!"

Output:

What's your name?
Alice
Hello, Alice!
Output Formatting

Using echo for formatted output:

# output.nim
let items = 5
let price = 12.5
let total = items * price

echo "Items: ", items
echo "Price per item: $", price
echo "Total: $", total.formatFloat(ffDecimal, 2)

Output:

Items: 5
Price per item: $12.5
Total: $62.50
Arithmetic Operators

Basic calculations in Nim:

# calc.nim
let a = 15
let b = 4

echo "a + b = ", a + b
echo "a - b = ", a - b
echo "a * b = ", a * b
echo "a / b = ", a / b
echo "a div b = ", a div b
echo "a mod b = ", a mod b

Output:

a + b = 19
a - b = 11
a * b = 60
a / b = 3.75
a div b = 3
a mod b = 3
Conditional Logic

Grading system with conditions:

# grade.nim
proc getGrade(score: int): string =
  if score >= 90:
    result = "A"
  elif score >= 80:
    result = "B"
  elif score >= 70:
    result = "C"
  elif score >= 60:
    result = "D"
  else:
    result = "F"

echo getGrade(85) # Output: B
Loop Examples

Various loop patterns:

# loops.nim
# For loop with range
for i in 1..3:
  echo "For loop: ", i

# While loop
var j = 1
while j <= 3:
  echo "While loop: ", j
  j += 1

# Iterating over array
let colors = ["red", "green", "blue"]
for color in colors:
  echo "Color: ", color

Output:

For loop: 1
For loop: 2
For loop: 3
While loop: 1
While loop: 2
While loop: 3
Color: red
Color: green
Color: blue
Complete Program

A simple number guessing game:

# guess.nim
import random, strutils

randomize()
let secret = rand(1..100)
var attempts = 0

echo "Guess a number between 1 and 100"

while true:
  let input = readLine(stdin)
  if not input.isDigit():
    echo "Please enter a number"
    continue
  let guess = parseInt(input)
  attempts += 1
  if guess == secret:
    echo "Correct! You guessed it in ", attempts, " attempts"
    break
  elif guess < secret:
    echo "Too low!"
  else:
    echo "Too high!"
